    L. for sharing implementation in component, just use composables in vue
    ex. create usePageSectionWithAnchor compsable file
    import { onBeforeUnmount, onMounted, ref } from "vue";

    interface PageSection {
      id: string;
      offsetTop: number;
    }

    export const usePageSectionWithAnchor = (initialSections: PageSection[]) => {
      const pageSections = initialSections as PageSection[];
      const activePageSectionId = ref("");

      const updatePageSectionTop = (): void => {
        pageSections.forEach((section) => {
          const sectionElement = document.getElementById(section.id);
          section.offsetTop = sectionElement?.offsetTop ?? 0;
        });
      };

      const handleScroll = (): void => {
        const scrollPosition = window.scrollY;

        if (!pageSections) {
          return;
        }

        for (let i = pageSections.length - 1; i >= 0; i--) {
          const section = pageSections[i];

          if (scrollPosition < section.offsetTop) {
            continue;
          }

          if (activePageSectionId.value === section.id) {
            break;
          }

          activePageSectionId.value = section.id;
          break;
        }
      };

      onMounted(() => {
        updatePageSectionTop();
        window.addEventListener("scroll", handleScroll);
      });

      onBeforeUnmount(() => {
        window.removeEventListener("scroll", handleScroll);
      });

      return {
        updatePageSectionTop
      };
    };

    export default defineComponent({

41. constructor like c# or other OO native language, in typescript multiple constructors are not allowed so,
    export class Money {
      public currency: string | undefined;
      public symbol: string | undefined;
      public amount: number = 0;

      constructor() {
        this.currency = "GBP";
        this.symbol = "£";
      }

      constructor(amount: number = 0) {
        this.currency = "GBP";
        this.symbol = "£";
        this.amount = amount;
      }
    }

    above will turn into this,
    export class Money {
      public currency: string;
      public symbol: string;
      public amount: number;

      constructor(amount: number = 0) {
        this.currency = "GBP";
        this.symbol = "£";
        this.amount = amount;
      }
    }
